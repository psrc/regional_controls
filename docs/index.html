Income Distribution 
{
  const syn = synthetic[selected_forecast];
  const pums_vals = pums.income;
  const labels = income_labels;
  const trace_pums = {
    x: labels,
    y: pums_vals,
    name: "PUMS " + base_year,
    type: "bar",
    marker: { color: "#054B6E" }
  };
  const trace_syn = {
    x: labels,
    y: syn.income,
    name: "Synthetic " + forecast_year,
    type: "bar",
    marker: { color: "#F05A28" }
  };
  const layout = {
    barmode: "group",
    height: 400,
    yaxis: { title: "Proportion of Households", tickformat: ".1%" },
    legend: { orientation: "h", y: 1.12 },
    margin: { t: 30, b: 80 }
  };
  const div = DOM.element("div");
  Plotly.newPlot(div, [trace_pums, trace_syn], layout, {responsive: true});
  return div;
}

Income Distribution — Change from Base Year 
{
  const labels = income_labels;
  const base = pums_counts.income;
  const traces = selected_forecasts.map((fc, i) => {
    const syn = synthetic_counts[fc].income;
    const deltas = syn.map((v, j) => v - base[j]);
    return {
      x: labels,
      y: deltas,
      name: fc,
      type: "bar",
      marker: { color: psrc_colors[i % psrc_colors.length] }
    };
  });
  const layout = {
    barmode: "group",
    height: 400,
    yaxis: { title: "Change from Base Year" },
    legend: { orientation: "h", y: 1.15 },
    margin: { t: 30, b: 80 },
    shapes: [{ type: "line", x0: 0, x1: 1, xref: "paper", y0: 0, y1: 0, line: { color: "#999", width: 1, dash: "dot" } }]
  };
  const div = DOM.element("div");
  Plotly.newPlot(div, traces, layout, {responsive: true});
  return div;
}
